Compressing and Decompressing String in Java using GZIP streams

<div class='post-body entry-content' id='post-body-5411597660880334924' itemprop='description articleBody'>
Following code can be used to compress or decompress any data in Java using the GZIP streams - java's built in zip streams.

<pre style="font-family: Andale Mono, Lucida Console, Monaco, fixed, monospace; color: #000000; background-color: #eee;font-size: 12px;border: 1px dashed #999999;line-height: 14px;padding: 5px; overflow: auto; width: 100%"><code>    public static String compress(String str) throws IOException {
        if (str == null &#124;&#124; str.length() == 0) {
            return str;
        }
        System.out.println(&quot;String length : &quot; + str.length());
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        GZIPOutputStream gzip = new GZIPOutputStream(out);
        gzip.write(str.getBytes());
        gzip.close();
        String outStr = out.toString(&quot;ISO-8859-1&quot;);
        System.out.println(&quot;Output String lenght : &quot; + outStr.length());
        return outStr;
     }
    
    public static String decompress(String str) throws IOException {
        if (str == null &#124;&#124; str.length() == 0) {
            return str;
        }
        System.out.println(&quot;Input String length : &quot; + str.length());
        GZIPInputStream gis = new GZIPInputStream(new ByteArrayInputStream(str.getBytes(&quot;ISO-8859-1&quot;)));
        BufferedReader bf = new BufferedReader(new InputStreamReader(gis, &quot;ISO-8859-1&quot;));
        String outStr = &quot;&quot;;
        String line;
        while ((line=bf.readLine())!=null) {
          outStr += line;
        }
        System.out.println(&quot;Output String lenght : &quot; + outStr.length());
        return outStr;
     }
  
     public static void main(String[] args) throws IOException {
